REM now loop all possible command characters
allChars = "abcdefghijklmnopqrstuvwxyz1234567890!@#$%^&*()-_=+`~,<.>/?;:'"
allChars += '"'
allChars += "[{]}\|]"

for (( i=0; i<${#allChars}; i++ )); do
  nextUp = "${allChars:$i:1}"
  Q CTRL ${nextUp}
  Q CTRL ALT ${nextUp}
  Q CTRL SHIFT ${nextUp}
  Q CTRL ALT SHIFT ${nextUp}
  Q GUI ${nextUp}
  Q GUI CTRL ${nextUp}
  Q GUI ALT ${nextUp}
  Q GUI SHIFT ${nextUp}
  Q GUI CTRL ALT ${nextUp}
  Q GUI CTRL SHIFT ${nextUp}
  Q GUI CTRL ALT SHIFT ${nextUp}
done
